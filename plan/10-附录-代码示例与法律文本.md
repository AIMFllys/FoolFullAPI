## A. AI API 调用封装

```typescript
// server/src/services/aiService.ts
import axios from 'axios';

type ModelType = 'kimi' | 'deepseek';

interface AiMessage {
  role: 'system' | 'user' | 'assistant';
  content: string;
}

const MODEL_CONFIG: Record<ModelType, { model: string; maxTokens: number; temperature: number }> = {
  kimi: {
    model: process.env.KIMI_MODEL || 'kimi',
    maxTokens: 500,
    temperature: 0.4,
  },
  deepseek: {
    model: process.env.DEEPSEEK_MODEL || 'deepseek-chat',
    maxTokens: 150,
    temperature: 0.3,
  },
};

export async function callAiApi(model: ModelType, messages: AiMessage[]): Promise<string> {
  const config = MODEL_CONFIG[model];
  try {
    const response = await axios.post(
      process.env.QINIU_API_URL!,
      { model: config.model, messages, max_tokens: config.maxTokens, temperature: config.temperature },
      {
        headers: { 'Authorization': `Bearer ${process.env.QINIU_API_KEY}`, 'Content-Type': 'application/json' },
        timeout: 15000,
      }
    );
    return response.data.choices[0]?.message?.content?.trim() || '';
  } catch (error) {
    console.error(`[AI:${model}] 调用失败:`, error);
    throw error;
  }
}
```

## B. 6 步聊天服务核心逻辑

```typescript
// server/src/services/chatService.ts
import { STEP_PROMPTS } from '../config/prompts';
import { FALLBACK_REPLIES, EASTER_EGG_MESSAGES } from '../utils/fallbacks';
import { sanitizeUserMessage } from '../utils/sanitizer';
import { callAiApi } from './aiService';

const LEAK_KEYWORDS = ['愚人节', '整蛊', '玩笑', 'prank', 'april fool', '骗你', '恶作剧'];

export async function processChat(
  userMessage: string,
  currentStep: number,
  firstMessage: string | null,
  isIdeMode: boolean
): Promise<{ reply: string; isEasterEgg: boolean; modelUsed: string }> {
  const sanitized = sanitizeUserMessage(userMessage);

  // Step 2: 模板硬编码
  if (currentStep === 2) {
    const taskDesc = (firstMessage || sanitized).slice(0, 20).replace(/[？?。！!]/g, '');
    return {
      reply: `📊 任务处理中...\n\n✅ Step 1/3: 需求理解 — 已完成\n✅ Step 2/3: 方案构建 — 已完成\n⏳ Step 3/3: 生成「${taskDesc}」— 进行中 (67%)...\n\n预计剩余时间：8秒`,
      isEasterEgg: false,
      modelUsed: 'template',
    };
  }

  // Step 6: 硬编码彩蛋
  if (currentStep === 6) {
    const idx = Math.floor(Math.random() * EASTER_EGG_MESSAGES.length);
    return { reply: EASTER_EGG_MESSAGES[idx], isEasterEgg: true, modelUsed: 'hardcode' };
  }

  // Step 1, 5 → Kimi | Step 3, 4 → DeepSeek
  const model = (currentStep === 1 || currentStep === 5) ? 'kimi' : 'deepseek';
  const contextMessage = firstMessage || sanitized;

  const coreConstraint = `【核心约束】你当前处于步骤 ${currentStep}，必须严格遵循步骤要求。不能回答用户的实际问题，不能透露整蛊意图。`;
  const messages = [
    { role: 'system' as const, content: coreConstraint },
    { role: 'system' as const, content: STEP_PROMPTS[currentStep] },
    { role: 'user' as const, content: isIdeMode ? `[IDE_MODE] ${contextMessage}` : contextMessage },
  ];

  try {
    let reply = await callAiApi(model, messages);

    // Step 5 后处理：确保包含中断标记
    if (currentStep === 5) {
      if (!reply.includes('⚠️') && !reply.includes('[系统提示]')) {
        const cutPoint = Math.min(reply.length, 200);
        reply = reply.slice(0, cutPoint) + '\n\n⚠️ [系统提示] 检测到当前会话为愚人节特别版本，内容生成已暂停。';
      }
    }

    // 关键词泄露检查
    for (const kw of LEAK_KEYWORDS) {
      if (reply.toLowerCase().includes(kw)) {
        return { reply: FALLBACK_REPLIES[currentStep], isEasterEgg: false, modelUsed: model };
      }
    }

    return { reply, isEasterEgg: false, modelUsed: model };
  } catch {
    return { reply: FALLBACK_REPLIES[currentStep], isEasterEgg: currentStep === 6, modelUsed: 'fallback' };
  }
}
```

## C. 前端聊天核心逻辑

```typescript
// client/src/pages/ChatPage.tsx (核心片段)
const STEP_DELAYS: Record<number, [number, number]> = {
  1: [3000, 5000], 2: [2000, 3000], 3: [1000, 2000],
  4: [1000, 2000], 5: [4000, 6000], 6: [0, 0],
};

const sendMessage = async () => {
  if (!input.trim() || isLoading) return;
  setMessages(prev => [...prev, { id: crypto.randomUUID(), role: 'user', content: input.trim() }]);
  setInput(''); setIsLoading(true);
  setMessages(prev => [...prev, { id: 'thinking', role: 'ai', content: '', isTyping: true }]);

  try {
    const res = await chatApi.send(input.trim());
    const [min, max] = STEP_DELAYS[res.step] || [1000, 2000];
    await new Promise(r => setTimeout(r, min + Math.random() * (max - min)));

    setMessages(prev => {
      const filtered = prev.filter(m => m.id !== 'thinking');
      return [...filtered, { id: crypto.randomUUID(), role: 'ai', content: res.reply, isEasterEgg: res.isEasterEgg }];
    });
    if (res.isEasterEgg) triggerConfetti();
  } catch { /* 错误处理 */ }
  finally { setIsLoading(false); }
};
```
